## countNodes

Counts all nodes in a tree including nested children.

### Signature

```typescript
function countNodes(nodes: WfNode[]): number
```

### Parameters

<ParamField path="nodes" type="WfNode[]" required>
  Array of root nodes to count
</ParamField>

### Returns

Total number of nodes including all descendants.

### Example

```typescript
import { WorkFlowyClient, collectAllRoots, countNodes } from 'workflowy-scraper'

const client = new WorkFlowyClient('your-session-id')
const data = await client.fetchInitData()
const roots = collectAllRoots(data)

const total = countNodes(roots)
console.log(`Total nodes: ${total}`)
```

## findAuxTree

Finds an auxiliary (shared/team) tree by its display name.

### Signature

```typescript
function findAuxTree(data: InitData, name: string): AuxTreeInfo | undefined
```

### Parameters

<ParamField path="data" type="InitData" required>
  The initialization data from WorkFlowy API
</ParamField>

<ParamField path="name" type="string" required>
  Display name of the auxiliary tree to find
</ParamField>

### Returns

The matching `AuxTreeInfo` object, or `undefined` if not found.

### Example

```typescript
import { WorkFlowyClient, findAuxTree } from 'workflowy-scraper'

const client = new WorkFlowyClient('your-session-id')
const data = await client.fetchInitData()

const teamTree = findAuxTree(data, 'Team Projects')
if (teamTree) {
  console.log('Found team tree:', teamTree.rootProject.nm)
  console.log('Share ID:', teamTree.shareId)
}
```

## listAuxTreeNames

Returns the display names of all auxiliary (shared/team) trees.

### Signature

```typescript
function listAuxTreeNames(data: InitData): string[]
```

### Parameters

<ParamField path="data" type="InitData" required>
  The initialization data from WorkFlowy API
</ParamField>

### Returns

Array of tree names with HTML tags stripped.

### Example

```typescript
import { WorkFlowyClient, listAuxTreeNames } from 'workflowy-scraper'

const client = new WorkFlowyClient('your-session-id')
const data = await client.fetchInitData()

const names = listAuxTreeNames(data)
console.log('Shared trees:', names)
// Output: ['Team Projects', 'Client Docs', 'Personal Share']
```
